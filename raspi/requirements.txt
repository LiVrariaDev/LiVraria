# NFCカードリーダー
# ============================================================================
pyscard==2.1.1  # smartcardモジュール（PC/SC経由でカードリーダーと通信）

# ============================================================================
# オプション: nfcpyを使う場合
# ============================================================================
# nfcpy==1.0.4  # NFC通信ライブラリ
# ndeflib==0.3.3  # NFCデータ形式
# libusb1==3.3.1  # USB通信
# pyserial==3.5  # シリアル通信
# pyDes==2.0.1  # DES暗号化

# システム要件（apt経由でインストール）

# sudo apt-get install -y pcscd pcsc-tools libpcsclite-dev

# sudo systemctl enable pcscd
#
